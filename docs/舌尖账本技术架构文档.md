## 一、技术可行性说明

### 1.1 技术选型确认

| 方案 | 可行性 | 说明 |
|------|--------|------|
| **.NET WebAPI + MSSQL** | ✅ 可行 | 经典B/S架构，EF Core完美支持MSSQL，食物库可直接对接 |
| **uniapp + 微信小程序** | ✅ 可行 | uniapp原生支持微信小程序生态，编译产物为小程序原生代码 |
| **iOS APP（原生）** | ✅ 可行 | Swift/SwiftUI开发，原生性能，Apple生态标准方案 |
| **uniapp + 微信小程序** | ✅ 可行 | uniapp原生支持微信小程序生态，编译产物为小程序原生代码 |

### 1.2 关键技术疑虑解答

**Q：.NET WebAPI能否为移动端提供可靠接口？**
A：完全可行。ASP.NET Core WebAPI支持RESTful设计，配合JWT认证，可为小程序、APP提供统一接口服务。EF Core 9.0对SQL Server支持成熟，性能有保障。

**Q：iOS原生开发与WebAPI配合是否有问题？**
A：无问题。iOS原生应用通过URLSession发起HTTP/HTTPS请求调用RESTful API是标准做法。JWT认证机制与平台无关，任何客户端均可使用。

**Q：微信小程序和iOS APP数据是否互通？**
A：互通。两端共用同一后端API和数据库，用户通过UnionID（微信绑定）或手机号（APP绑定）关联同一账号，数据实时同步。

**Q：多端数据如何保证实时同步？**
A：采用"共享数据库 + 统一API"架构，小程序与APP共用同一后端和数据库，通过用户认证体系（UnionID/手机号）关联，数据天然一致。

## 三、数据库设计

### 3.1 设计原则

- **食物库复用**：已有MSSQL食物库直接对接，不迁移数据
- **业务库独立**：新建业务数据库（FoodRecords, Users等），与食物库分离
- **跨库查询**：通过EF Core配置多DbContext，实现跨库关联查询

```sql
CREATE TABLE FoodCategories (
    id                    INT IDENTITY(1,1) PRIMARY KEY,

CREATE INDEX IX_FoodCategories_CateId ON FoodCategories(cate_id);
```

```swift
// Core/Network/APIClient.swift
import Foundation

final class APIClient {
    static let shared = APIClient()

    private let baseURL = "https://api.chime.com"
    private let session: URLSession

    private init() {
        let config = URLSessionConfiguration.default
        config.timeoutIntervalForRequest = 30
        self.session = URLSession(configuration: config)
    }

    func request<T: Decodable>(
        _ endpoint: APIEndpoint,
        responseType: T.Type
    ) async throws -> T {
        var request = URLRequest(url: URL(string: baseURL + endpoint.path)!)
        request.httpMethod = endpoint.method.rawValue
        request.setValue("application/json", forHTTPHeaderField: "Content-Type")

        if let token = KeychainHelper.shared.getToken() {
            request.setValue("Bearer \(token)", forHTTPHeaderField: "Authorization")
        }

        if let body = endpoint.body {
            request.httpBody = try JSONEncoder().encode(body)
        }

        let (data, response) = try await session.data(for: request)

        guard let httpResponse = response as? HTTPURLResponse else {
            throw NetworkError.invalidResponse
        }

        let apiResponse = try JSONDecoder().decode(APIResponse<T>.self, from: data)

        guard apiResponse.code == 200 else {
            if apiResponse.code == 401 {
                NotificationCenter.default.post(name: .userDidLogout, object: nil)
            }
            throw NetworkError.apiError(apiResponse.message)
        }

        return apiResponse.data
    }
}

// APIEndpoint.swift
enum HTTPMethod: String {
    case GET, POST, PUT, DELETE
}

struct APIEndpoint {
    let path: String
    let method: HTTPMethod
    let body: Encodable?

    static func login(phone: String, code: String) -> APIEndpoint {
        APIEndpoint(
            path: "/api/auth/phone-login",
            method: .POST,
            body: ["phoneNumber": phone, "verificationCode": code]
        )
    }

    static func getFoods(keyword: String?, page: Int) -> APIEndpoint {
        var path = "/api/foods?page=\(page)&pageSize=20"
        if let keyword = keyword {
            path += "&keyword=\(keyword)"
        }
        return APIEndpoint(path: path, method: .GET, body: nil)
    }

    // ... 其他端点
}

// APIResponse.swift
struct APIResponse<T: Decodable>: Decodable {
    let code: Int
    let message: String
    let data: T
}
```

```javascript
    }
    return cache.data
```

```csharp
// Infrastructure/Caching/RedisCacheService.cs
public interface ICacheService
{
    Task<T?> GetAsync<T>(string key);
    Task SetAsync<T>(string key, T value, TimeSpan? expiry = null);
    Task RemoveAsync(string key);
    Task<bool> ExistsAsync(string key);
}

public class RedisCacheService : ICacheService
{
    private readonly IDatabase _db;
    private readonly IConnectionMultiplexer _redis;

    public RedisCacheService(IConnectionMultiplexer redis)
    {
        _redis = redis;
        _db = redis.GetDatabase();
    }

    public async Task<T?> GetAsync<T>(string key)
    {
        var value = await _db.StringGetAsync(key);
        if (value.IsNullOrEmpty) return default;

        return JsonSerializer.Deserialize<T>(value!);
    }

    public async Task SetAsync<T>(string key, T value, TimeSpan? expiry = null)
    {
        var json = JsonSerializer.Serialize(value);
        await _db.StringSetAsync(key, json, expiry);
    }

    public async Task RemoveAsync(string key)
    {
        await _db.KeyDeleteAsync(key);
    }
}
```

```csharp
// 缓存空值防止穿透
public async Task<Food?> GetFoodByIdAsync(long id)
{
    var cacheKey = $"food:{id}";
    var cached = await _cache.GetAsync<Food>(cacheKey);

    if (cached != null)
        return cached;

    // 从数据库查询
    var food = await _foodRepository.GetByIdAsync(id);

    // 即使查询结果为空，也缓存空值（短期），防止穿透
    if (food == null)
    {
        await _cache.SetAsync(cacheKey, new Food(), TimeSpan.FromMinutes(5));
    }
    else
    {
        await _cache.SetAsync(cacheKey, food, TimeSpan.FromHours(24));
    }

    return food;
}
```

```csharp
// 使用随机TTL避免同时失效
public async Task<List<FoodCategory>> GetCategoriesAsync()
{
    var cacheKey = "food:categories";
    var cached = await _cache.GetAsync<List<FoodCategory>>(cacheKey);

    if (cached != null) return cached;
```

### 8.2 文件上传流程

```csharp
// Api/Controllers/UploadsController.cs
[HttpPost("photo")]
[RequestSizeLimit(10 * 1024 * 1024)]  // 10MB限制
public async Task<IActionResult> UploadPhoto(IFormFile file)
{
    // 1. 验证文件
    var allowedExtensions = new[] { ".jpg", ".jpeg", ".png" };
    var extension = Path.GetExtension(file.FileName).ToLowerInvariant();
    if (!allowedExtensions.Contains(extension))
        return BadRequest("仅支持jpg、jpeg、png格式");

    // 2. 生成文件路径
    var userId = GetCurrentUserId();
    var dateFolder = DateTime.Now.ToString("yyyyMMdd");
    var fileName = $"{userId}_{Guid.NewGuid()}{extension}";
    var relativePath = $"photos/{dateFolder}/{fileName}";

    // 3. 存储文件（开发环境存本地，生产环境存OSS）
    var fileUrl = await _fileStorageService.UploadAsync(relativePath, file.OpenReadStream());

    // 4. 返回文件URL
    return Ok(new { url = fileUrl, localPath = relativePath });
}
```

### 8.3 文件存储服务

```csharp
// Application/Services/IFileStorageService.cs
public interface IFileStorageService
{
    Task<string> UploadAsync(string path, Stream stream);
    Task<bool> DeleteAsync(string path);
    string GetAccessUrl(string path);
}

// 开发环境实现
public class LocalFileStorageService : IFileStorageService
{
    private readonly IWebHostEnvironment _env;

    public async Task<string> UploadAsync(string path, Stream stream)
    {
        var fullPath = Path.Combine(_env.ContentRootPath, "uploads", path);
        var directory = Path.GetDirectoryName(fullPath);
        Directory.CreateDirectory(directory!);

        await using var fileStream = new FileStream(fullPath, FileMode.Create);
        await stream.CopyToAsync(fileStream);

        return $"/uploads/{path}";
    }
}

// 生产环境实现（阿里云OSS）
public class OssFileStorageService : IFileStorageService
{
    private readonly IOssClient _ossClient;
    private readonly string _bucketName;

    public async Task<string> UploadAsync(string path, Stream stream)
    {
        var objectName = path.Replace("\\", "/");
        await _ossClient.PutObjectAsync(_bucketName, objectName, stream);

        // 返回CDN地址（如果配置了CDN）
        return $"https://cdn.chime.com/{objectName}";
    }
}
```

### 9.4 全局异常处理

```csharp
// Middleware/ExceptionHandlingMiddleware.cs
public class ExceptionHandlingMiddleware
{
    private readonly RequestDelegate _next;
    private readonly ILogger<ExceptionHandlingMiddleware> _logger;

    public async Task InvokeAsync(HttpContext context)
    {
        try
        {
            await _next(context);
        }
        catch (Exception ex)
        {
            _logger.Error(ex, "未处理的异常: {Path}", context.Request.Path);
            await HandleExceptionAsync(context, ex);
        }
    }

    private static async Task HandleExceptionAsync(HttpContext context, Exception exception)
    {
        context.Response.ContentType = "application/json";
        context.Response.StatusCode = exception switch
        {
            ArgumentException => 400,
            UnauthorizedAccessException => 401,
            KeyNotFoundException => 404,
            _ => 500
        };

        var response = new
        {
            code = context.Response.StatusCode,
            message = exception.Message,
            traceId = context.TraceIdentifier
        };

        await context.Response.WriteAsJsonAsync(response);
    }
}
```
